[20] Profile - 기본값
아래처럼 props 값인 job 에서 개발자라는 기본 값을 지정하여서 태그에 props를 사용하는 실습을 해보았다.

// Profile.jsx 
// job 생략 시 "개발자" 기본값 사용 (= 로 지정)
const Profile = ({ name, job = "개발자" }) => {
  return (
    <div>
      <h3>{name}</h3>
      <p>💼 직업: {job}</p>
    </div>
  )
}
export default Profile

// App.jsx
import Profile from './components/Profile.jsx'

const App = () => {
  return (
    <>
      <Profile name="지니" job="프론트엔드" />   {/* job 넘김 */}
      <Profile name="이디자인" />                {/* job 생략 → 개발자(기본값) */}
    </>
  )
}
export default App
